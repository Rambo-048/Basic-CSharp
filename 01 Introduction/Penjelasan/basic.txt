==== INPUT ====
- {0} ==> Sebuah Placeholder yang nantinya akan diisi dengan Isi dari Variabel
- $   ==> String Interpolation
- "..." + nama + "..."   ==> Concatenate


using System;
using System.Collections.Generic;  
using System.Linq;   
using System.Text;   
using System.Threading.Tasks; 

namespace HelloHello
{
	class Program
	{
		static void Main(Strings[] args) 
		{
			Console.WriteLine("Even You Don't Have Support From Some People In Your Life, You Must Keep Moving");
			
			Console.WriteLine("This is INPUT Function");
			Console.Write("Insert Your Name : ");

			String nama = Console.ReadLine();   <--- Fungsi Input


			Console.WriteLine("Hello, {0} !!", nama);                     // <-- Menggunakan Placeholder
			Console.WriteLine($"Hello, {nama} Good Nite!!");              // <-- Teknik String Interpolasi
			Console.WriteLine("Hello, " + nama + " Selamat Datang!");     // <-- Concatenate			
		}
	}
}



==== PERBEDAAN WRITE DAN WRITELINE ====
1. Write(); ---> Mencetak Ke layar tanpa membuat Baris Baru
2. WriteLine(); ---> Mencetak Ke Layar dengan Membuat Baris Baru

using System;
using System.Collections.Generic;   
using System.Linq;   
using System.Text;   
using System.Threading.Tasks; 

namespace HelloHello
{
	class Program
	{

		{
			Console.WriteLine("Perbedaan Write dan Write Line");

			Console.Write("Heath Ledger");
			Console.WriteLine();
			Console.Write("Alamat : ");
			Console.Write("Unknown Street");
			Console.WriteLine();
		}
	}
}
